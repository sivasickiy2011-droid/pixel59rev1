    """
    –ü—Ä–æ–≤–µ—Ä—è–µ—Ç –∫–∞—á–µ—Å—Ç–≤–æ –∏–∑–≤–ª–µ—á–µ–Ω–Ω–æ–≥–æ –∫–æ–Ω—Ç–µ–Ω—Ç–∞.

    """
    if not content or len(content) < min_length:
        return False
    
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º —Å–æ–æ—Ç–Ω–æ—à–µ–Ω–∏–µ –±—É–∫–≤ –∫ –¥—Ä—É–≥–∏–º —Å–∏–º–≤–æ–ª–∞–º

    if letters < len(content) * 0.5:  # –ú–∏–Ω–∏–º—É–º 50% –±—É–∫–≤
        return False
    
    # –ü—Ä–æ–≤–µ—Ä—è–µ–º –Ω–∞–ª–∏—á–∏–µ –ø—Ä–µ–¥–ª–æ–∂–µ–Ω–∏–π

    if sentences < 2:
        return False
    
    return True
```
